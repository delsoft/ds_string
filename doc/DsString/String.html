# File lib/ds_string/core_ext.rb, line 31
def times num
  ret = ''
  num.times do 
    if block_given?
      tmp = ret if ret != ''
      ret += yield(self, tmp) 
    else
      ret += self 
    end
  end
  ret
end

# File lib/ds_string/core_ext.rb, line 13
def to_bool
  return true if self == true || self =~ (/(true|t|yes|y|1)$/)
  return false if self == false || self == '' || self == nil || self =~ (/(false|f|no|n|0)$/)
  raise ArgumentError.new("invalid value for Boolean: \"#{self}\"")
end
